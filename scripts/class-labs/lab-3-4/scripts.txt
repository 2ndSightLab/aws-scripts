aws cloudformation create-stack --stack-name Lab-3-3-S3 --template-body file://1-s3-bucket.yaml --parameters ParameterKey=bucketname,ParameterValue=<unique bucket name>

echo "File that is unencrypted when uploaded." >> unencrypted.txt

echo "File that is encrypted when uploaded." >> encrypted.txt

aws kms create-key

aws s3 cp unencrypted.txt s3://<your bucket name>
aws s3 cp --sse aws:kms --sse-kms-key-id <keyid from previous step> encrypted.txt s3://<your bucket name>
